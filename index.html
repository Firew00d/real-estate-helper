/** * 국토부 부동산 실거래가 API 프록시 (Cloudflare Worker) * ------------------------------------------------------ * - API 키를 서버에 숨겨 브라우저로 노출되지 않게 함 * - CORS 자동 해결 * - XML → JSON 변환 * - 최근 N개월 자동 수집 (병렬 호출) * - 건물명/면적 기반 후필터링 * * 사용 엔드포인트 (GET): * /?sigunguCode=11680 * &buildingType=아파트 (아파트 | 오피스텔 | 연립다세대) * &tradeType=매매 (매매 | 전월세) * &months=6 (기본 6) * &buildingName=래미안 (선택: 건물명 필터) * &area=59 (선택: 전용면적 ±20% 필터) * * 응답 예시: * { * "ok": true, * "buildingType": "아파트", * "tradeType": "매매", * "sigunguCode": "11680", * "months": ["202511","202510", ...], * "totalCount": 43, * "filteredCount": 5, * "avg_sale_price": 185000, // 만원 (매매만) * "avg_deposit": null, // 만원 (전월세만) * "avg_monthly_rent": null, // 만원 (전월세만) * "transactions": [ * { "date":"2025-11", "type":"매매", "name":"래미안...", "area":84.5, "floor":"15", "price":180000, "deposit":null, "monthlyRent":null, "dong":"역삼동", "jibun":"123" }, * ... * ], * "source": "국토교통부 공공데이터포털" * } */ const API_BASE = 'https://apis.data.go.kr/1613000'; /* ========================================================== 🔑 국토교통부 공공데이터포털 서비스키 (Decoding 버전) ========================================================== Cloudflare Worker의 Secret(이름: MOLIT_API_KEY)에서 읽어옵니다. 대시보드 → Worker → Settings → Variables and Secrets 에서 등록하세요. ========================================================== */ // 건물종류 + 거래유형 → 엔드포인트 매핑 const ENDPOINTS = { '아파트': { '매매': 'RTMSDataSvcAptTradeDev/getRTMSDataSvcAptTradeDev', '전월세': 'RTMSDataSvcAptRent/getRTMSDataSvcAptRent' }, '오피스텔': { '매매': 'RTMSDataSvcOffiTrade/getRTMSDataSvcOffiTrade', '전월세': 'RTMSDataSvcOffiRent/getRTMSDataSvcOffiRent' }, '연립다세대': { '매매': 'RTMSDataSvcRHTrade/getRTMSDataSvcRHTrade', '전월세': 'RTMSDataSvcRHRent/getRTMSDataSvcRHRent' }, '단독다가구': { '매매': 'RTMSDataSvcSHTrade/getRTMSDataSvcSHTrade', '전월세': 'RTMSDataSvcSHRent/getRTMSDataSvcSHRent' }, }; // CORS 헤더 — 필요 시 origin 화이트리스트로 제한 가능 const CORS_HEADERS = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Methods': 'GET, POST, OPTIONS', 'Access-Control-Allow-Headers': 'Content-Type, x-api-key, anthropic-version', 'Access-Control-Max-Age': '86400', }; export default { async fetch(request, env) { // Preflight if (request.method === 'OPTIONS') { return new Response(null, { headers: CORS_HEADERS }); } const url = new URL(request.url); // ────────────────────────────────────────────────────────── // Route: /claude → Anthropic API 프록시 (POST) // ────────────────────────────────────────────────────────── if (url.pathname === '/claude') { return handleClaudeProxy(request, env); } // ────────────────────────────────────────────────────────── // Route: / (기본) → 국토부 실거래가 API 프록시 (GET) // ────────────────────────────────────────────────────────── if (request.method !== 'GET') { return json({ ok: false, error: 'Only GET allowed on this route' }, 405); } const sigunguCode = url.searchParams.get('sigunguCode'); const buildingType = url.searchParams.get('buildingType') || '아파트'; const tradeType = url.searchParams.get('tradeType') || '매매'; const months = Math.max(1, Math.min(12, parseInt(url.searchParams.get('months') || '6', 10))); const buildingName = (url.searchParams.get('buildingName') || '').trim(); const areaStr = url.searchParams.get('area'); const area = areaStr ? parseFloat(areaStr) : null; // 입력 검증 if (!/^\d{5}$/.test(sigunguCode || '')) { return json({ ok: false, error: 'sigunguCode는 5자리 숫자여야 합니다.' }, 400); } const endpoint = ENDPOINTS[buildingType]?.[tradeType]; if (!endpoint) { return json({ ok: false, error: `지원하지 않는 조합: ${buildingType}/${tradeType}` }, 400); } if (!env.MOLIT_API_KEY) { return json({ ok: false, error: 'MOLIT_API_KEY Secret이 설정되지 않았습니다. Worker 대시보드에서 Secret을 등록하세요.' }, 500); } // 최근 N개월 YYYYMM 생성 const yyyymmList = lastNMonths(months); // 병렬 호출 const results = await Promise.all( yyyymmList.map(ymd => fetchMonth(endpoint, sigunguCode, ymd, env.MOLIT_API_KEY)) ); // 병합 및 정규화 const allItems = []; let errorsCount = 0; for (const r of results) { if (r.error) { errorsCount++; continue; } allItems.push(...r.items); } // 정규화 + 정렬 (최신순) const normalized = allItems .map(item => normalizeItem(item, buildingType, tradeType)) .filter(Boolean) .sort((a, b) => (b._sortKey || '').localeCompare(a._sortKey || '')); // 필터링 (건물명, 면적) let filtered = normalized; if (buildingName) { const kw = buildingName.replace(/\s+/g, '').toLowerCase(); filtered = filtered.filter(t => (t.name || '').replace(/\s+/g, '').toLowerCase().includes(kw)); } if (area && !isNaN(area)) { const lo = area * 0.8, hi = area * 1.2; filtered = filtered.filter(t => t.area && t.area >= lo && t.area <= hi); } // 평균가 계산 const stats = computeStats(filtered.length > 0 ? filtered : normalized, tradeType); return json({ ok: true, buildingType, tradeType, sigunguCode, months: yyyymmList, totalCount: normalized.length, filteredCount: filtered.length, apiErrors: errorsCount, usedFilter: filtered.length > 0 && (buildingName || area), ...stats, transactions: (filtered.length > 0 ? filtered : normalized).slice(0, 20) .map(({ _sortKey, ...rest }) => rest), source: '국토교통부 공공데이터포털', }); } }; // ────────────────────────────────────────────────────────────── // Helpers // ────────────────────────────────────────────────────────────── function json(obj, status = 200) { return new Response(JSON.stringify(obj, null, 2), { status, headers: { 'Content-Type': 'application/json; charset=utf-8', ...CORS_HEADERS }, }); } function lastNMonths(n) { const out = []; const now = new Date(); // 실거래가 신고는 계약일로부터 30일이라 "이번달"은 데이터가 적음 → 지난달부터 시작 for (let i = 1; i <= n; i++) { const d = new Date(now.getFullYear(), now.getMonth() - i, 1); const ym = `${d.getFullYear()}${String(d.getMonth() + 1).padStart(2, '0')}`; out.push(ym); } return out; } async function fetchMonth(endpoint, lawdCd, dealYmd, serviceKey) { const params = new URLSearchParams({ serviceKey, // 공공데이터포털에서 발급 (Decoding 버전) LAWD_CD: lawdCd, DEAL_YMD: dealYmd, pageNo: '1', numOfRows: '500', // 월 단위 거래 많은 강남/마포 등 고려 }); const url = `${API_BASE}/${endpoint}?${params.toString()}`; try { const resp = await fetch(url, { method: 'GET', headers: { 'Accept': 'application/xml' }, cf: { cacheTtl: 3600, cacheEverything: true }, // 1시간 엣지 캐시 }); if (!resp.ok) return { error: `HTTP ${resp.status}`, items: [] }; const xml = await resp.text(); const items = parseXmlItems(xml); return { items }; } catch (e) { return { error: String(e), items: [] }; } } /** * 국토부 API XML은 ...... 구조. * Worker에는 DOMParser가 없어서 정규식 기반 가벼운 파서를 사용. * 필드는 숫자/영문 태그 이름으로만 구성되어 있고 중첩이 없으므로 안전. */ function parseXmlItems(xml) { // SERVICE_KEY_IS_NOT_REGISTERED 같은 에러 응답 감지 if (/(?!00)/.test(xml) && !/00<\/resultCode>/.test(xml)) { const msg = (xml.match(/([^<]+)<\/resultMsg>/) || [])[1] || ''; if (msg && !/정상/.test(msg) && !/NORMAL/i.test(msg)) { // 정상이 아니면 빈 배열 return []; } } const items = []; const itemRegex = /([\s\S]*?)<\/item>/g; let m; while ((m = itemRegex.exec(xml)) !== null) { const inner = m[1]; const obj = {}; // 단일 레벨 태그만 파싱 const tagRegex = /<([A-Za-z][A-Za-z0-9]*)>([\s\S]*?)<\/\1>/g; let t; while ((t = tagRegex.exec(inner)) !== null) { obj[t[1]] = t[2].trim(); } items.push(obj); } return items; } /** * 건물종류별 응답 필드를 공통 스키마로 정규화 */ function normalizeItem(item, buildingType, tradeType) { const year = item.dealYear || ''; const month = (item.dealMonth || '').padStart(2, '0'); const day = (item.dealDay || '').padStart(2, '0'); if (!year || !month) return null; const date = `${year}-${month}`; const sortKey = `${year}${month}${day}`; // 건물명 let name = ''; if (buildingType === '아파트') name = item.aptNm || ''; if (buildingType === '오피스텔') name = item.offiNm || ''; if (buildingType === '연립다세대') name = item.mhouseNm || ''; // 단독/다가구는 건물명이 없음 → houseType("단독", "다가구")을 표시명으로 사용 if (buildingType === '단독다가구') name = item.houseType || '단독/다가구'; // 면적 (단독/다가구는 연면적 totalFloorAr, 나머지는 전용면적 excluUseAr) let area = null; if (buildingType === '단독다가구') { area = item.totalFloorAr ? parseFloat(item.totalFloorAr) : null; } else { area = item.excluUseAr ? parseFloat(item.excluUseAr) : null; } // 금액 파싱 (콤마 제거) const toInt = v => { if (!v) return null; const n = parseInt(String(v).replace(/[^\d-]/g, ''), 10); return isNaN(n) ? null : n; }; const base = { date, type: tradeType, name: name || '(미상)', area, floor: item.floor || null, dong: item.umdNm || '', jibun: item.jibun || '', buildYear: item.buildYear ? parseInt(item.buildYear, 10) : null, _sortKey: sortKey, }; if (tradeType === '매매') { base.price = toInt(item.dealAmount); // 만원 base.deposit = null; base.monthlyRent = null; } else { base.price = null; base.deposit = toInt(item.deposit); // 만원 base.monthlyRent = toInt(item.monthlyRent); // 만원 // 월세가 0이면 실질 전세 if (base.monthlyRent === 0) base.monthlyRent = null; } return base; } function computeStats(items, tradeType) { if (tradeType === '매매') { const prices = items.map(i => i.price).filter(n => n && n > 0); return { avg_sale_price: avg(prices), median_sale_price: median(prices), min_sale_price: prices.length ? Math.min(...prices) : null, max_sale_price: prices.length ? Math.max(...prices) : null, avg_deposit: null, avg_monthly_rent: null, }; } else { const leaseOnly = items.filter(i => (i.monthlyRent || 0) === 0 || i.monthlyRent === null) .map(i => i.deposit).filter(n => n && n > 0); const withRent = items.filter(i => i.monthlyRent && i.monthlyRent > 0); return { avg_sale_price: null, avg_deposit: avg(items.map(i => i.deposit).filter(n => n && n > 0)), avg_lease_deposit: avg(leaseOnly), // 순수 전세 평균 보증금 avg_monthly_rent: avg(withRent.map(i => i.monthlyRent)), avg_monthly_deposit: avg(withRent.map(i => i.deposit).filter(n => n && n > 0)), }; } } function avg(arr) { if (!arr || arr.length === 0) return null; return Math.round(arr.reduce((s, v) => s + v, 0) / arr.length); } function median(arr) { if (!arr || arr.length === 0) return null; const s = [...arr].sort((a, b) => a - b); const m = Math.floor(s.length / 2); return s.length % 2 ? s[m] : Math.round((s[m - 1] + s[m]) / 2); } // ────────────────────────────────────────────────────────────── // Anthropic API 프록시 (/claude 경로) // ────────────────────────────────────────────────────────────── /** * 브라우저 → Worker(/claude) → Anthropic API로 요청을 그대로 전달. * - API 키는 Worker Secret(ANTHROPIC_API_KEY)에서 읽어 HTML에 노출되지 않음 * - CORS 자동 해결 * - 요청 body는 그대로 Anthropic에 전달 (model, messages, tools, max_tokens 등 투명 전달) */ async function handleClaudeProxy(request, env) { if (request.method !== 'POST') { return json({ error: 'POST required for /claude' }, 405); } if (!env.ANTHROPIC_API_KEY) { return json({ error: 'ANTHROPIC_API_KEY Secret이 설정되지 않았습니다. Worker 대시보드에서 Secret을 등록하세요.' }, 500); } let body; try { body = await request.text(); } catch (e) { return json({ error: 'Invalid request body' }, 400); } try { const resp = await fetch('https://api.anthropic.com/v1/messages', { method: 'POST', headers: { 'Content-Type': 'application/json', 'x-api-key': env.ANTHROPIC_API_KEY, 'anthropic-version': '2023-06-01', }, body, }); // Anthropic 응답을 그대로 passthrough (단, CORS 헤더는 우리 것으로) const text = await resp.text(); return new Response(text, { status: resp.status, headers: { 'Content-Type': 'application/json; charset=utf-8', ...CORS_HEADERS, }, }); } catch (e) { return json({ error: `Upstream error: ${String(e)}` }, 502); } }
